**1. CSS Custom Properties (:root)**

```css
:root {
  /* Z-index scale */
  --z-scanlines: 10;
  --z-vignette: 20;
  --z-reflection: 21;

  /* Monochrome phosphor variants (OKLCH at 65% lightness) */
  --phosphor-green: oklch(65% 0.25 145);
  --phosphor-amber: oklch(65% 0.20 85);
  --phosphor-cyan: oklch(65% 0.20 195);
  --phosphor-red: oklch(65% 0.25 25);
  --phosphor-white: oklch(65% 0 0);
  --phosphor-orange: oklch(65% 0.22 55);

  /* RGB fallbacks for older browsers */
  --phosphor-green-fallback: rgb(0, 255, 136);
  --phosphor-amber-fallback: rgb(255, 191, 0);
  --phosphor-cyan-fallback: rgb(0, 255, 255);
  --phosphor-red-fallback: rgb(255, 77, 77);
  --phosphor-white-fallback: rgb(229, 229, 229);
  --phosphor-orange-fallback: rgb(255, 153, 51);
}
```
