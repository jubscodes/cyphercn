  **React 19 ref migration pattern:**
  Replace `React.forwardRef<RefType, PropsType>(({...props}, ref) => ...)` with a plain function that accepts `ref?: React.Ref<RefType>` as a destructured prop. Remove `.displayName` assignments (not needed without forwardRef). Keep all other functionality identical.

  Example transformation:
  ```tsx
  // BEFORE (React 18 forwardRef)
  const Foo = React.forwardRef<HTMLDivElement, FooProps>(({className, ...props}, ref) => (
    <div ref={ref} className={className} {...props} />
  ));
  Foo.displayName = "Foo";

  // AFTER (React 19 ref prop)
  function Foo({className, ref, ...props}: FooProps & { ref?: React.Ref<HTMLDivElement> }) {
    return <div ref={ref} className={className} {...props} />;
  }
  ```
