```javascript
// Bayer 4x4 matrix (normalized 0-16)
const BAYER_4x4 = [
  [0, 8, 2, 10],
  [12, 4, 14, 6],
  [3, 11, 1, 9],
  [15, 7, 13, 5]
];

// Message handler
self.onmessage = async function(e) {
  const { imageBitmap, width, height, algorithm } = e.data;

  // Create offscreen canvas
  const canvas = new OffscreenCanvas(width, height);
  const ctx = canvas.getContext('2d');

  // Draw image
  ctx.drawImage(imageBitmap, 0, 0, width, height);

  // Get image data
  const imageData = ctx.getImageData(0, 0, width, height);

  // Apply dithering
  const dithered = applyDither(imageData, algorithm);

  // Put back
  ctx.putImageData(dithered, 0, 0);

  // Convert to blob and send back
  const blob = await canvas.convertToBlob({ type: 'image/png' });
  self.postMessage({ blob });
};

function applyDither(imageData, algorithm) {
  const { data, width, height } = imageData;
  const output = new ImageData(width, height);

  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      const i = (y * width + x) * 4;

      // Convert to grayscale
      const gray = 0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2];

      let threshold;
      if (algorithm === 'bayer') {
        threshold = (BAYER_4x4[y % 4][x % 4] / 16) * 255;
      } else if (algorithm === 'halftone') {
        // Simple halftone pattern
        const px = x % 4;
        const py = y % 4;
        threshold = ((px + py * 4) / 16) * 255;
      } else {
        // Noise dithering
        threshold = Math.random() * 255;
      }

      const value = gray > threshold ? 255 : 0;

      output.data[i] = output.data[i + 1] = output.data[i + 2] = value;
      output.data[i + 3] = data[i + 3]; // Preserve alpha
    }
  }

  return output;
}
```

```typescript
"use client";

import * as React from "react";
import { cn } from "@/lib/utils";
import "./styles/cyberpunk.css";

export type DitherAlgorithm = "bayer" | "halftone" | "noise";

export interface DitheredImageProps extends React.HTMLAttributes<HTMLDivElement> {
  src: string;
  alt: string;
  algorithm?: DitherAlgorithm;
  width?: number;
  height?: number;
  glow?: boolean;
  fallback?: React.ReactNode;
}

export function DitheredImage({
  src,
  alt,
  algorithm = "bayer",
  width,
  height,
  glow = false,
  fallback,
  className,
  ...props
}: DitheredImageProps) {
  const canvasRef = React.useRef<HTMLCanvasElement>(null);
  const [loading, setLoading] = React.useState(true);
  const [error, setError] = React.useState(false);

  React.useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    setLoading(true);
    setError(false);

    const img = new Image();
    img.crossOrigin = "anonymous";

    img.onload = async () => {
      const w = width || img.naturalWidth;
      const h = height || img.naturalHeight;
      canvas.width = w;
      canvas.height = h;

      // Try OffscreenCanvas with worker
      if (typeof OffscreenCanvas !== "undefined" && typeof createImageBitmap !== "undefined") {
        try {
          const bitmap = await createImageBitmap(img);
          const worker = new Worker("/workers/dither-worker.js");

          worker.onmessage = async (e) => {
            const { blob } = e.data;
            const url = URL.createObjectURL(blob);
            const ditheredImg = new Image();
            ditheredImg.onload = () => {
              const ctx = canvas.getContext("2d");
              if (ctx) {
                ctx.drawImage(ditheredImg, 0, 0);
              }
              URL.revokeObjectURL(url);
              setLoading(false);
            };
            ditheredImg.src = url;
            worker.terminate();
          };

          worker.onerror = () => {
            // Fallback to main thread
            processOnMainThread(canvas, img, algorithm);
            setLoading(false);
            worker.terminate();
          };

          worker.postMessage({ imageBitmap: bitmap, width: w, height: h, algorithm });
        } catch {
          processOnMainThread(canvas, img, algorithm);
          setLoading(false);
        }
      } else {
        // Fallback for browsers without OffscreenCanvas
        processOnMainThread(canvas, img, algorithm);
        setLoading(false);
      }
    };

    img.onerror = () => {
      setError(true);
      setLoading(false);
    };

    img.src = src;
  }, [src, algorithm, width, height]);

  if (error) {
    return fallback ? <>{fallback}</> : (
      <div className={cn("cyphercn text-xs text-foreground/50 border border-foreground/30 flex items-center justify-center", className)}
           style={{ width, height }}>
        [IMAGE ERROR]
      </div>
    );
  }

  return (
    <div className={cn("cyphercn relative inline-block", glow && "phosphor-glow", className)} {...props}>
      <canvas
        ref={canvasRef}
        className={cn("block", loading && "opacity-0")}
        aria-label={alt}
        role="img"
      />
      {loading && (
        <div className="absolute inset-0 flex items-center justify-center text-xs text-foreground/50">
          [PROCESSING...]
        </div>
      )}
    </div>
  );
}

// Main thread fallback
function processOnMainThread(canvas: HTMLCanvasElement, img: HTMLImageElement, algorithm: DitherAlgorithm) {
  const ctx = canvas.getContext("2d");
  if (!ctx) return;

  ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
  const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);

  const BAYER_4x4 = [
    [0, 8, 2, 10],
    [12, 4, 14, 6],
    [3, 11, 1, 9],
    [15, 7, 13, 5]
  ];

  const { data, width, height } = imageData;

  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      const i = (y * width + x) * 4;
      const gray = 0.299 * data[i] + 0.587 * data[i + 1] + 0.114 * data[i + 2];

      let threshold: number;
      if (algorithm === "bayer") {
        threshold = (BAYER_4x4[y % 4][x % 4] / 16) * 255;
      } else if (algorithm === "halftone") {
        const px = x % 4;
        const py = y % 4;
        threshold = ((px + py * 4) / 16) * 255;
      } else {
        threshold = Math.random() * 255;
      }

      const value = gray > threshold ? 255 : 0;
      data[i] = data[i + 1] = data[i + 2] = value;
    }
  }

  ctx.putImageData(imageData, 0, 0);
}
```
