```typescript
"use client";

import * as React from "react";
import { cn } from "@/lib/utils";
import "./styles/cyberpunk.css";

export interface GaugeProps extends React.HTMLAttributes<HTMLDivElement> {
  value: number;           // Current value
  max?: number;            // Maximum value (default 100)
  size?: number;           // SVG size in pixels (default 80)
  strokeWidth?: number;    // Arc stroke width (default 4)
  label?: string;          // Label below gauge
  showValue?: boolean;     // Show percentage in center (default true)
  glow?: boolean;          // Phosphor glow effect
  variant?: "default" | "danger" | "warning";  // Color variants
}

export function Gauge({
  value,
  max = 100,
  size = 80,
  strokeWidth = 4,
  label,
  showValue = true,
  glow = false,
  variant = "default",
  className,
  ...props
}: GaugeProps) {
  const percentage = Math.min(100, Math.max(0, (value / max) * 100));
  const radius = (size - strokeWidth) / 2;
  const circumference = 2 * Math.PI * radius;
  const offset = circumference - (percentage / 100) * circumference;

  const variantColors = {
    default: "stroke-current",
    danger: "stroke-red-500",
    warning: "stroke-yellow-500",
  };

  return (
    <div
      className={cn("cyphercn flex flex-col items-center gap-1", className)}
      {...props}
    >
      <svg
        width={size}
        height={size}
        viewBox={`0 0 ${size} ${size}`}
        className={cn(glow && "phosphor-glow")}
      >
        {/* Background circle */}
        <circle
          cx={size / 2}
          cy={size / 2}
          r={radius}
          fill="none"
          stroke="currentColor"
          strokeWidth={strokeWidth}
          opacity={0.2}
        />
        {/* Progress arc */}
        <circle
          cx={size / 2}
          cy={size / 2}
          r={radius}
          fill="none"
          className={cn(variantColors[variant], "transition-all duration-300")}
          strokeWidth={strokeWidth}
          strokeDasharray={circumference}
          strokeDashoffset={offset}
          strokeLinecap="round"
          transform={`rotate(-90 ${size / 2} ${size / 2})`}
          style={{
            // Respect reduced motion
            transitionDuration: "var(--gauge-transition, 300ms)",
          }}
        />
        {/* Center value */}
        {showValue && (
          <text
            x={size / 2}
            y={size / 2 + 4}
            textAnchor="middle"
            className="fill-current text-xs tabular-nums"
            style={{ fontFamily: "var(--font-mono, monospace)" }}
          >
            {Math.round(percentage)}%
          </text>
        )}
      </svg>
      {label && (
        <span className="text-xs uppercase tracking-wider text-foreground/70">
          {label}
        </span>
      )}
    </div>
  );
}
```

```typescript
export interface SystemMonitorProps extends React.HTMLAttributes<HTMLDivElement> {
  cpu: number;             // 0-100 percentage
  memory: number;          // 0-100 percentage
  cpuMax?: number;         // Default 100
  memoryMax?: number;      // Default 100
  title?: string;          // Panel title (default "SYSTEM")
  glow?: boolean;
  border?: "single" | "double" | "ascii";
}

// Terminal border styles (same as TerminalLog)
const terminalBorderStyles = {
  single: "border border-foreground",
  double: "border-[3px] border-double border-foreground",
  ascii: "border-2 border-foreground",
};

export function SystemMonitor({
  cpu,
  memory,
  cpuMax = 100,
  memoryMax = 100,
  title = "SYSTEM",
  glow = false,
  border = "single",
  className,
  ...props
}: SystemMonitorProps) {
  const borderStyle = terminalBorderStyles[border];

  // Determine variants based on thresholds
  const cpuVariant = cpu > 90 ? "danger" : cpu > 70 ? "warning" : "default";
  const memVariant = memory > 90 ? "danger" : memory > 70 ? "warning" : "default";

  return (
    <div
      className={cn(
        "cyphercn relative w-fit",
        borderStyle,
        glow && "phosphor-glow",
        className
      )}
      {...props}
    >
      {/* Title */}
      {title && (
        <div className="absolute -top-[0.6em] left-3 bg-background px-1 text-xs uppercase tracking-wider text-foreground">
          {title}
        </div>
      )}

      {/* Gauges */}
      <div className="flex items-center gap-4 px-4 py-3">
        <Gauge
          value={cpu}
          max={cpuMax}
          label="CPU"
          variant={cpuVariant}
          size={64}
        />
        <Gauge
          value={memory}
          max={memoryMax}
          label="MEM"
          variant={memVariant}
          size={64}
        />
      </div>
    </div>
  );
}
```

```typescript
export interface NetworkMonitorProps extends React.HTMLAttributes<HTMLDivElement> {

  glow?: boolean;
  border?: "single" | "double" | "ascii";
  unit?: string;           // Unit label (default "Mbps")
}

export function NetworkMonitor({
  upload,
  download,
  uploadMax = 100,
  downloadMax = 100,
  title = "NETWORK",
  glow = false,
  border = "single",
  unit = "Mbps",
  className,
  ...props
}: NetworkMonitorProps) {

  return (
    <div
```
